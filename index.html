setSize(1000, 750); setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); setLocationRelativeTo(null); currentMonth = YearMonth.of(2026, 4); // --- Πάνω Μέρος: Πλοήγηση --- JPanel navPanel = new JPanel(new BorderLayout()); navPanel.add(nextBtn, BorderLayout.EAST); // --- Κέντρο: Πίνακας --- String[] columns = {"Δευτέρα", "Τρίτη", "Τετάρτη", "Πέμπτη", "Παρασκευή", "Σάββατο", "Κυριακή"}; tableModel = new DefaultTableModel(null, columns) { @Override public boolean isCellEditable(int r, int c) { return false; } }; calendarTable = new JTable(tableModel); calendarTable.setRowHeight(90); calendarTable.getTableHeader().setReorderingAllowed(false); calendarTable.setDefaultRenderer(Object.class, new ShiftRenderer()); // --- Κάτω Μέρος: Στατιστικά & Κουμπί --- JPanel bottomPanel = new JPanel(new GridLayout(1, 2)); infoLabel = new JLabel("Πενθήμερα Μήνα: -", SwingConstants.LEFT); infoLabel.setFont(new Font("Tahoma", Font.BOLD, 14)); tableModel.setRowCount(0); LocalDate firstOfMonth = currentMonth.atDay(1); int dayOfWeek = firstOfMonth.getDayOfWeek().getValue(); // Υπολογισμός πόσα ΣΚ έχει ο μήνας (Πενθήμερα βάσει ημερολογίου) int weekends = countWeekendsInMonth(currentMonth); for (int d = 1; d <= ym.lengthOfMonth(); d++) { DayOfWeek dw = ym.atDay(d).getDayOfWeek(); if (dw == DayOfWeek.SATURDAY) count++; // Μετράμε τα Σάββατα ως δείκτη εβδομάδων } return count; } private void calculateMonthlyStats() { int totalNights = 0; int workedWeekends = 0; for (int d = 1; d <= currentMonth.lengthOfMonth(); d++) { LocalDate date = currentMonth.atDay(d); long diff = java.time.temporal.ChronoUnit.DAYS.between(anchorDate, date); int cycleIdx = (int) (diff % 25); if (cycleIdx < 0) cycleIdx += 25; // Ορισμός Βαρδιών (ίδιος με τον Renderer) String[] shiftNames = {"ΒΡ", "ΑΔ", "ΠΡ", "ΣΥ", "ΑΠ", "ΒΡ", "ΑΔ", "08-16", "ΣΥ", "ΑΠ", "ΒΡ", "ΑΔ", "ΠΡ", "ΣΥ", "ΑΠ", "ΒΡ", "ΑΔ", "ΠΡ", "ΣΥ", "ΑΠ", "ΒΡ", "ΑΔ", "ΠΡ", "ΣΥ", "ΑΠ 3-11"}; } } JOptionPane.showMessageDialog(this, "Στατιστικά για " + monthLabel.getText() + ":\n" + "----------------------------------\n" + "• Νυχτερινά (ΒΡ): " + totalNights + "\n" + "• Εργάσιμα Σάββατα/Κυριακές: " + workedWeekends + "\n" + "----------------------------------\n" + "Σημείωση: Τα πενθήμερα εξαρτώνται από το λογιστήριο.", "Καταχώρηση Μήνα", JOptionPane.INFORMATION_MESSAGE); @Override public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int col) { Component c = super.getTableCellRendererComponent(table, value, isSelected, hasFocus, row, col); if (value == null || value.toString().isEmpty()) { // Χρώμα για τα κενά κελιά του ΣΚ if (col >= 5) c.setBackground(new Color(240, 240, 240)); else c.setBackground(Color.WHITE); setText(""); return c; } int day = (int) value; LocalDate date = currentMonth.atDay(day); long diff = java.time.temporal.ChronoUnit.DAYS.between(anchorDate, date); int cycleIdx = (int) (diff % 25); if (cycleIdx < 0) cycleIdx += 25; fg = Color.GRAY; } // Ειδικό περίγραμμα ή χρώμα αν είναι Σάββατο (col 5) ή Κυριακή (col 6) if (col >= 5 && shift.equals("ΑΔ")) { bg = new Color(180, 190, 200); // Πιο σκούρο γκρι στα ρεπό του ΣΚ } c.setBackground(bg); c.setForeground(fg); // Αν είναι ΣΚ, βάζουμε ένα border για να φαίνεται η διαφορά if (col >= 5) { ((JComponent)c).setBorder(BorderFactory.createLineBorder(new Color(0, 102, 204), 1)); } else { ((JComponent)c).setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY, 1)); } setText("
" + day + "
" + shift + "
" + locations[cycleIdx] + "
"); return c; } } public static void main(String[] args) { SwingUtilities.invokeLater(() -> new ShiftCalendar().setVisible(true)); } }
